295 + this._model

318function _MinchinWeb_Atlas_::Pop() {
319 // Returns the top rated pair as an array and removes the pair from the model
320
321 // If the two tiles returned are equal, pop another one
322 local KeepTrying = true;
323 local Test;
324 while (KeepTrying == true) {
325 Test = this._pairs.Pop();
326 if ((Test == null) || (Test[0] != Test[1])) {
327 KeepTrying = false;
328 }
329 }
330 return Test;

372function _MinchinWeb_Atlas_::ApplyTrafficModel(StartTile, StartPriority, EndTile, EndPriority, Model) {
373 // Given the start and end points, applies the traffic model and returns
374 // the weighting
375 // Smaller weightings are considered better
376
377 switch (Model) {
378 case ModelType.ONE_D :
379 return AIMap.DistanceMax(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
380 break;
381 case ModelType.DISTANCE_MANHATTAN :
382 return AIMap.DistanceManhattan(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
383 break;
384 case ModelType.DISTANCE_SHIP :
385 return _MinchinWeb_Marine_.DistanceShip(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
386 break;
387 case ModelType.DISTANCE_AIR :
388 return AIMap.DistanceSquare(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
389 break;
390 case ModelType.DISTANCE_NONE :
391 return (1.0 / (StartPriority.tofloat() + EndPriority.tofloat()));
392 break;
393 case ModelType.ONE_OVER_T_SQUARED :
394 return ((AIMap.DistanceManhattan(StartTile, EndTile) * AIMap.DistanceManhattan(StartTile, EndTile)) / (StartPriority.tofloat() + EndPriority.tofloat()));
395 break;
396 default:
397 return "ERROR: Bad ModelType. Was supplied: " + Model;
398 break;
399 }
