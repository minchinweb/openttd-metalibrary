159 if (this._end != null) {

220function _MinchinWeb_LW_::Slope(Slope, ThirdQuadrant = false) {
221 if (_MinchinWeb_Extras_.AbsFloat(Slope) > _MinchinWeb_C_.Infinity()) {
222 AILog.Warning("Slope is capped at " + _MinchinWeb_C_.Infinity() + ", you provided " + Slope + ".");
223 this._slope = _MinchinWeb_C_.Infinity();
224 } else if (_MinchinWeb_Extras_.AbsFloat(Slope) < (1.0 / _MinchinWeb_C_.Infinity())) {
225 AILog.Warning("Slope is capped at 1/" + _MinchinWeb_C_.Infinity() + ", you provided " + Slope + ".");
226 this._slope = (1.0 / _MinchinWeb_C_.Infinity());
227 } else {
228 this._slope = Slope;
229 }

231 if (ThirdQuadrant == false) {

284function _MinchinWeb_LW_::Walk() {
285 if (this._past_end == true) {
286 return this._current_tile;
287 }
288
289 if ((AIMap.DistanceManhattan(this._current_tile, AIMap.GetTileIndex(this._x.tointeger(), this._y.tointeger())) == 1 ) && _MinchinWeb_Extras_.WithinFloat(this._start_x.tofloat(), this._end_x.tofloat(), this._x.tointeger()) &&_MinchinWeb_Extras_.WithinFloat(this._start_y.tofloat(), this._end_y.tofloat(), this._y.tointeger())) {
290 this._current_tile = AIMap.GetTileIndex(this._x.tointeger(), this._y.tointeger());
291// _MinchinWeb_Log_.Note("Linewalker output " + AIMap.GetTileX(this._current_tile) + "," + AIMap.GetTileY(this._current_tile) + " from " + this._x + "," + this._y, 7);
292 return this._current_tile;
293 }
294
295 // Infinity assumed to be 10,000
296 local multiplier = 0.0;
297
298 // We need to find the value, such that MAX(ABS(∆x, m∆x)) == 1
299 // Therefore, our multiplier is MIN(ABS(1, 1/m))
300 multiplier = _MinchinWeb_Extras_.MinAbsFloat(1.0, (1.0 / this._slope) );
301
302 local NewX = 0.0;
303 local NewY = 0.0;
304 NewX = this._x + multiplier * this._dir_x;
305 NewY = this._y + this._slope * multiplier * this._dir_x;
306// _MinchinWeb_Log_.Note("Linewalker new : " + NewX + "," + NewY, 7);
307
308 if (AIMap.DistanceManhattan(this._current_tile, AIMap.GetTileIndex(NewX.tointeger(), NewY.tointeger())) == 1 ) {
309 this._current_tile = AIMap.GetTileIndex(NewX.tointeger(), NewY.tointeger());
310 } else if (AIMap.DistanceManhattan(this._current_tile, AIMap.GetTileIndex(NewX.tointeger(), this._y.tointeger())) == 1 ) {
311 this._current_tile = AIMap.GetTileIndex(NewX.tointeger(), this._y.tointeger());
312 }
313
314 this._x = NewX;
315 this._y = NewY;
316
317 // Check that we're still within our bounding box
318// _MinchinWeb_Log_.Note(" " + this._start_x + " , " + this._end_x + " , " + this._x.tointeger() + " , " + this._start_y + " , " + this._end_y + " , " + this._y.tointeger(), 7);
319
320 if (!_MinchinWeb_Extras_.WithinFloat(this._start_x.tofloat(), this._end_x.tofloat(), this._x) || !_MinchinWeb_Extras_.WithinFloat(this._start_y.tofloat(), this._end_y.tofloat(), this._y)) {
321// _MinchinWeb_Log_.Note("Linewalker outside box " + this._start_x + " " + this._end_x + " " + this._x + " " + _MinchinWeb_Extras_.WithinFloat(this._start_x.tofloat(), this._end_x.tofloat(), this._x) + " : " + this._start_y + " " + this._end_y + " " + this._y + " " + (_MinchinWeb_Extras_.WithinFloat(this._start_y.tofloat(), this._end_y.tofloat(), this._y)), 6);
322 this._past_end = true;
323 return this._current_tile;
324 } else {
325// _MinchinWeb_Log_.Note("Linewalker output " + AIMap.GetTileX(this._current_tile) + "," + AIMap.GetTileY(this._current_tile), 6);
326 return this._current_tile;
327 }
