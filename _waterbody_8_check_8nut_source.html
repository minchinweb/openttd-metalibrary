190function _MinchinWeb_WBC_::_Estimate(self, cur_tile, cur_direction, goal_tiles) {
191 local min_cost = self._max_cost;
192 /* As estimate we multiply the lowest possible cost for a single tile with
193 * with the minimum number of tiles we need to traverse. */
194 foreach (tile in goal_tiles) {
195 min_cost = min(AIMap.DistanceManhattan(cur_tile, tile) * self._cost_per_tile * self._distance_penalty, min_cost);
196 }
197 return min_cost;
198}

200function _MinchinWeb_WBC_::_Neighbours(self, path, cur_node) {
201 /* self._max_cost is the maximum path cost, if we go over it, the path isn't valid. */
202 if (path.GetCost() >= self._max_cost) return [];
203 local tiles = [];
204
205 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
206 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
207 /* Check all tiles adjacent to the current tile. */
208 foreach (offset in offsets) {
209 local next_tile = cur_node + offset;
210 if (AIMarine.AreWaterTilesConnected(cur_node, next_tile)) {
211 tiles.push([next_tile, self._GetDirection(cur_node, next_tile)]);
212 }
213 }
214
217
218 return tiles;
219}

225function _MinchinWeb_WBC_::_GetDirection(from, to) {
226 if (AITile.GetSlope(to) == AITile.SLOPE_FLAT) return 0xFF;
227 if (from - to == 1) return 1;
228 if (from - to == -1) return 2;
229 if (from - to == AIMap.GetMapSizeX()) return 4;
230 if (from - to == -AIMap.GetMapSizeX()) return 8;
