233function _MinchinWeb_Lakes_::FindPath(iterations) {
234 // This is where the meat and potatoes is!
235 // See the diagram in the docs for how this works
236 if (iterations < 0) {
237 iterations = _MinchinWeb_C_.Infinity();
238 }
239 for (local i = 0; i < iterations; i++) {
240 local tick = AIController.GetTick();
241 if (_MinchinWeb_Array_.Compare1D(this._AGroup, [-1]) || _MinchinWeb_Array_.Compare1D(this._BGroup, [-1])) {
242 // one end is all on land
243 // no path is possible
244 _MinchinWeb_Log_.Note("Lakes failed to find a path. One end is all on land.", 5);
245 this._running = false;
246 return null;
247 }
248
249 // Get not only the groups the tiles are in, but all the tile groups
250 // that are connected
251 local AAllGroups = _AllGroups(this._AGroup);
252 _MinchinWeb_Log_.Note("AGroup: " + _MinchinWeb_Array_.ToString1D(this._AGroup, false) + " All A:" + _MinchinWeb_Array_.ToString1D(AAllGroups, false), 6);
253
254 foreach (Group in AAllGroups) {
255 if (_MinchinWeb_Array_.ContainedIn1D(this._BGroup, Group)) {
256 // If we have a connection, return 'true'
257 _MinchinWeb_Log_.Note("B Group (" + this._BGroup[0] + ")found in A All Groups!", 6);
258 this._running = false;
259 return true;
260 }
261 }
262
263 // No match (yet anyway...)
264 // Get all the open neighbours of A
265 local ANeighbours = array(0);
266 local AEdge = array(0);
267 local BAllGroups = array(0);
268 AAllGroups = _AllGroups(this._AGroup);
269 BAllGroups = _AllGroups(this._BGroup);
270
271 foreach (Group in AAllGroups){
272 ANeighbours = _MinchinWeb_Array_.Append(ANeighbours, this._open_neighbours[Group]);
273 }
274 foreach (neighbour in ANeighbours) {
275 AEdge.append(neighbour[0]);
276 }
277
278 // remove duplicates
279 AEdge = _MinchinWeb_Array_.RemoveDuplicates(AEdge);
280 _MinchinWeb_Log_.Note("A -- Edge: " + AEdge.len() + " Neighbours: " + ANeighbours.len(), 6);
281 _MinchinWeb_Log_.Note("A -- Edge: " + _MinchinWeb_Array_.ToStringTiles1D(AEdge), 7);
282
283 if (ANeighbours.len() > 0) {
284 // Get the tile from AEdge that is closest to B's
285 local BTileList = AIList();
286 foreach (group in BAllGroups) {
287 BTileList.AddList(this._group_tiles[group]);
288 }
289 local AEdgeHeap = this._heap_class();
290 foreach (edge in AEdge) {
291 AEdgeHeap.Insert(edge, _MinchinWeb_Extras_.MinDistance(edge, BTileList));
292 }
293
294 //_MinchinWeb_Log_.Note(" B -- Tiles: " + _MinchinWeb_Array_.ToStringTiles1D(BTileArray), 7);
295
296 // Process the tile's 4 neighbours x12
297 for (local j=0; j < 12; j++) {
298 if (AEdgeHeap.Count() > 0) {
299 local NextNeighbour = AEdgeHeap.Pop();
300 local AddedNeighbours = _AddNeighbour(NextNeighbour);
301 foreach (Tile in AddedNeighbours) {
302 if (Tile != null) {
303 AEdgeHeap.Insert(Tile, _MinchinWeb_Extras_.MinDistance(Tile, BTileList));
304 }
305 }
306 }
307 }
308 } else {
309 // With no 'open neighbours', there can be no more connections
310 this._running = false;
311 return null;
312 }
313
314 local BNeighbours = array(0);
315 local BEdge = array(0);
316 AAllGroups = _AllGroups(this._AGroup);
317 BAllGroups = _AllGroups(this._BGroup);
318
319 // Check to see if we have a connection (and can go home!)
320 _MinchinWeb_Log_.Note("BGroup: " + _MinchinWeb_Array_.ToString1D(this._AGroup, false) + " All B:" + _MinchinWeb_Array_.ToString1D(AAllGroups, false), 6);
321 foreach (Group in BAllGroups) {
322 if (_MinchinWeb_Array_.ContainedIn1D(this._AGroup, Group)) {
323 // If we have a connection, return 'true'
324 _MinchinWeb_Log_.Note("A Group (" + this._AGroup[0] + ") found in B All Groups!", 6);
325 this._running = false;
326 return true;
327 }
328 }
329
330 foreach (Group in BAllGroups) {
331 BNeighbours = _MinchinWeb_Array_.Append(BNeighbours, this._open_neighbours[Group]);
332 }
333 foreach (neighbour in BNeighbours) {
334 BEdge.append(neighbour[0]);
335 }
336 // remove duplicates
337 BEdge = _MinchinWeb_Array_.RemoveDuplicates(BEdge);
338 _MinchinWeb_Log_.Note("B -- Edge: " + BEdge.len() + " Neighbours: " + BNeighbours.len(), 6);
339 _MinchinWeb_Log_.Note("B -- Edge: " + _MinchinWeb_Array_.ToStringTiles1D(BEdge), 7);
340
341 if (BNeighbours.len() > 0) {
342 // Get the tile from AEdge that is closest to BEdge
343 local ATileList = AIList();
344 foreach (group in AAllGroups) {
345 ATileList.AddList(this._group_tiles[group]);
346 }
347 local BEdgeHeap = this._heap_class();
348 foreach (edge in BEdge) {
349 BEdgeHeap.Insert(edge, _MinchinWeb_Extras_.MinDistance(edge, ATileList));
350 }
351
352 //_MinchinWeb_Log_.Note(" A -- Tiles: " + _MinchinWeb_Array_.ToStringTiles1D(ATileArray), 7);
353
354 // Process the tile's 4 neighbours x12
355 for (local j=0; j < 12; j++) {
356 if (BEdgeHeap.Count() > 0) {
357 local NextNeighbour = BEdgeHeap.Pop();
358 local AddedNeighbours = _AddNeighbour(NextNeighbour);
359 foreach (Tile in AddedNeighbours) {
360 if (Tile != null) {
361 BEdgeHeap.Insert(Tile, _MinchinWeb_Extras_.MinDistance(Tile, ATileList));
362 }
363 }
364 }
365 }
366 } else {
367 // With no 'open neighbours', there can be no more connections
368 this._running = false;
369 return null;
370 }
371 _MinchinWeb_Log_.Note("B -- " + (AIController.GetTick() - tick) + " ticks.", 8);
372 }
373
374 // ran out of loops, we're still running
375 return false;

397function _MinchinWeb_Lakes_::AddPoint(myTileID) {
398 // _MinchinWeb_Log_.Note("Tile Area " + this._map.GetValue(myTileID) + " : " + this._map.Count() + " / " + this._areas.len(), 6);
399
400 switch (this._map.GetValue(myTileID)) {
401 case -2:
402 // tile is unset (i.e. in no group)
403 if (AITile.IsWaterTile(myTileID) == true) {
404 // add to _map if a water tile
405 local myArea = this._areas.len();
406 this._areas.append(myTileID);
407 this._open_neighbours.append([]);
408 this._connections.append([]);
409 this._map.SetValue(myTileID, myArea);
410 this._group_tiles.append(AIList());
411 this._group_tiles[myArea].AddItem(myTileID, myTileID);
412
413 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
414 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
415
416 foreach (offset in offsets) {
417 local next_tile = myTileID + offset;
418 if (AIMarine.AreWaterTilesConnected(myTileID, next_tile)) {
419 if (this._map.GetValue(next_tile) == -2) {
420 this._open_neighbours[myArea].append([myTileID, next_tile]);
421 } else if (this._map.GetValue(next_tile) == -1) {
422 // the tile has ceased to be land (somehow...)
423 this._map.SetValue(next_tile, -2);
424 this._open_neighbours[myArea].append([myTileID, next_tile]);
425 } else {
426 // register connection right now
427 local ConnectedArea = this._map.GetValue(next_tile);
428
429 this._connections[myArea].append(ConnectedArea);
430 this._connections[ConnectedArea].append(myArea);
431
432 local AllConnectedAreas = _AllGroups([ConnectedArea]);
433
434 // remove open neighbour from AllConnectedAreas to next_tile
435 foreach (ThisArea in AllConnectedAreas) {
436 for (local i=0; i < this._open_neighbours[ThisArea].len(); i++) {
437 if (this._open_neighbours[ThisArea][i][1] == myTileID) {
438 // we're looking for the reverse tile pair to the one we just tried to add
439 this._open_neighbours[ThisArea] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[ThisArea], i);
440 i--;
441 }
442 }
443 }
444 }
445 }
446 }
447
448 _MinchinWeb_Log_.Note(myArea + " : " + _MinchinWeb_Array_.ToStringTiles2D(this._open_neighbours[myArea], false), 7);
449 _MinchinWeb_Log_.Sign(myTileID, "L" + myArea, 8);
450 return myArea;
451 } else {
452 // land tile
453 this._map.SetValue(myTileID, -1);
454 // not added to _group_tiles
455 return -1;
456 }
457 case -1:
458 // land tile
459 return -1;
460 default:
461 // already in _map
462 return this._map.GetValue(myTileID);
463 }
464}

466function _MinchinWeb_Lakes_::_AllGroups(StartGroupArray) {
467 // this function starts with an array of starting groups
468 // starts at the beginning and for the first group, appends all connected
469 // groups to the end of the array
470 // does the same for the second and so on until the original end of the
471 // array
472 // next it compacts the resulting array by removing duplicates
473 // then it picks up where it left off in the (now larger) array and starts
474 // adding connections again
475 // this cycle keeps going until no more connections are added that aren't
476 // duplicates
477
478 local loops = 0;
479 local StartIndex = 0;
480 local ReturnGroup = StartGroupArray;
481 local NextStartIndex = 0;
482 local MoreAdded = true;
483
484 do {
485 //_MinchinWeb_Log_.Note("In AllGroups(), loop " + loops + ". start: " + StartIndex + " // " + _MinchinWeb_Array_.ToString1D(ReturnGroup, false), 6);
486 MoreAdded = false;
487 NextStartIndex = ReturnGroup.len();
488 for (local i = StartIndex; i < NextStartIndex; i++) {
489 if ((ReturnGroup[i] >= 0) && (this._connections[ReturnGroup[i]].len() > 0)) {
490 ReturnGroup = _MinchinWeb_Array_.Append(ReturnGroup, this._connections[ReturnGroup[i]]);
491 ReturnGroup = _MinchinWeb_Array_.RemoveDuplicates(ReturnGroup);
492 MoreAdded = true;
493 }
494 }
495 StartIndex = NextStartIndex;
496 loops++;
497 } while (MoreAdded == true)
498
499 return ReturnGroup;
500}

502function _MinchinWeb_Lakes_::_AddNeighbour(NextTile) {
503 // Start by finding out what area we're expanding
504 local ReturnTiles = array(0);
505 local OnwardTiles = array(0);
506 for (local i = 0; i < this._open_neighbours.len(); i++) {
507 for (local j = 0; j < this._open_neighbours[i].len(); j++) {
508 if (this._open_neighbours[i][j][0] == NextTile) {
509 OnwardTiles.append(this._open_neighbours[i][j][1]);
510 this._open_neighbours[i] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[i], j);
511 j--;
512 }
513 }
514 }
515 // remove duplicates
516 OnwardTiles = _MinchinWeb_Array_.RemoveDuplicates(OnwardTiles);
517 // remove onward tiles that are already in a tile group
518 local ConnectedGroups = _AllGroups([this._map[NextTile]]);
519 for (local i = 0; i < OnwardTiles.len(); i++) {
520 if (this._map[OnwardTiles[i]] != -2) {
521 // But only if we've already registered the connection
522 if (_MinchinWeb_Array_.ContainedIn1D(ConnectedGroups, this._map[OnwardTiles[i]])) {
523 OnwardTiles = _MinchinWeb_Array_.RemoveValueAt(OnwardTiles, i);
524 // Add remove from open neighbours
525 i--;
526 }
527 }
528
529 }
530 _MinchinWeb_Log_.Note("NextTile: " + _MinchinWeb_Array_.ToStringTiles1D([NextTile]) + " // Onward Tiles: " + _MinchinWeb_Array_.ToStringTiles1D(OnwardTiles), 8);
531
532 if (OnwardTiles.len() == 0) {
533 // if something broke, spit out useful debug information
534 _MinchinWeb_Log_.Note(" MinchinWeb.Lakes._AddNeighbour(): No source for " + _MinchinWeb_Array_.ToStringTiles1D([NextTile]), 8);
535 /*_MinchinWeb_Log_.Note(" this._open_neighbours");
536 for (local i = 0; i < this._open_neighbours.len(); i++) {
537 _MinchinWeb_Log_.Note(" [" + i + "] " + _MinchinWeb_Array_.ToString2D(this._open_neighbours[i]), 0);
538 }*/
539
540 return [null];
541 } else {
542 local FromGroup = this._map.GetValue(NextTile);
543 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
544 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
545
546 // Check that each neighbour is still attached by water to our from tile,
547 // and then add it to the FromGroup
548 // Then add possible neighbours to the open_neighbours list
549 foreach (OnwardTile in OnwardTiles) {
550 if (AIMarine.AreWaterTilesConnected(NextTile, OnwardTile)) {
551 this._map.SetValue(OnwardTile, FromGroup);
552 this._group_tiles[FromGroup].AddItem(OnwardTile, OnwardTile);
553 ReturnTiles.append(OnwardTile);
554 _MinchinWeb_Log_.Sign(OnwardTile, "L" + FromGroup, 8);
555 foreach (offset in offsets) {
556 local next_tile = OnwardTile + offset;
557 if (AIMarine.AreWaterTilesConnected(OnwardTile, next_tile) && (this._map.GetValue(next_tile) != this._map.GetValue(OnwardTile))) {
558 this._open_neighbours[FromGroup].append([OnwardTile, next_tile]);
559 }
560 }
561 }
562
563 // If more than one groups list this tile as an open neighbour,
564 // register the two groups are now linked
565 for (local i = 0; i < this._open_neighbours.len(); i++) {
566 for (local j = 0; j < this._open_neighbours[i].len(); j++) {
567 if (this._open_neighbours[i][j][1] == OnwardTile) {
568 local ActiveFromGroup = this._map.GetValue(this._open_neighbours[i][j][0]);
569 this._connections[FromGroup].append(ActiveFromGroup);
570 this._connections[ActiveFromGroup].append(FromGroup);
571 this._open_neighbours[i] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[i], j);
572 j--;
573
574 // remove duplicates
575 local oldActiveFromGroup = this._connections[ActiveFromGroup];
576 local oldFromGroup = this._connections[FromGroup];
577 this._connections[ActiveFromGroup] = _MinchinWeb_Array_.RemoveDuplicates(this._connections[ActiveFromGroup]);
578 this._connections[FromGroup] = _MinchinWeb_Array_.RemoveDuplicates(this._connections[FromGroup]);
579 //_MinchinWeb_Log_.Note(" Connections: " + FromGroup + " : " + _MinchinWeb_Array_.ToString1D(oldFromGroup, false) + "-> " + _MinchinWeb_Array_.ToString1D(this._connections[FromGroup], false), 7);
580 //_MinchinWeb_Log_.Note(" Connections: " + ActiveFromGroup + " : " + _MinchinWeb_Array_.ToString1D(oldActiveFromGroup, false) + "-> " + _MinchinWeb_Array_.ToString1D(this._connections[ActiveFromGroup], false), 7);
581 }
582 }
583 }
584 }
585 }
586 return ReturnTiles;
