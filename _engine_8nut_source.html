103function _MinchinWeb_Engine_::Rate2(EngineID, Cargo, TargetCapacity, MaxSpend = 0) {
104 // Designed to Run as a validator
105 // Given the EngineID, it will score them; higher is better
106 // Score = [(Capacity in Cargo) * Reliability * Speed]
107 // * CapacityFactor
108 // / [ (Purchase Price / Life) + (Running Costs) ]
109 //
110 // Note: Cargo doesn't fully work yet. Capacity is measured in the default cargo.
111
112 // only buildable engines
113 if (AIEngine.IsBuildable(EngineID)) {
114 // pass
115 } else {
116 return -1 * _MinchinWeb_C_.Infinity();
117 }
118
119 // only return vehicles under our hard cost limit
120 if ((MaxSpend > 0) && (AIEngine.GetPrice(EngineID) > MaxSpend)) {
121 return -1 * _MinchinWeb_C_.Infinity();
122 }
123
124 // only those that can carry our cargo
125 if (AIEngine.CanRefitCargo(EngineID, Cargo)) {
126 // pass
127 } else {
128 return -1 * _MinchinWeb_C_.Infinity();
129 }
130
131 local my_capacity = AIEngine.GetCapacity(EngineID);
132 // will be 0..1
133 local capacity_factor = 0;
134 if (my_capacity == TargetCapacity) {
135 capacity_factor = 1.0;
136 } else if (my_capacity > TargetCapacity) {
137 capacity_factor = TargetCapacity.tofloat() / my_capacity.tofloat();
138 } else {
139 // my_capacity < TargetCapacity
140 capacity_factor = my_capacity.tofloat() / TargetCapacity.tofloat();
141 }
142
143 // AIEngine.GetMaxAge() returns a value in (calendar) days
144 local max_age = AIEngine.GetMaxAge(EngineID).tofloat() / 365.0;
145 local annualize_purchase = AIEngine.GetPrice(EngineID).tofloat() / max_age;
146
147 local costs = 0.0001 + annualize_purchase + AIEngine.GetRunningCost(EngineID).tofloat();
148 // AIEngine.GetMaxSpeed() can be divided by 27 to get tiles per day
149 local benefits = (
150 0.0001
151 + AIEngine.GetCapacity(EngineID).tofloat()
152 * (AIEngine.GetReliability(EngineID).tofloat() / 100)
153 * (AIEngine.GetMaxSpeed(EngineID).tofloat() / 27 * 365)
154 );
155
156 local score = benefits * capacity_factor * 1000 / costs;
157 score = score.tointeger();
158 _MinchinWeb_Log_.Note(
159 "Engine.Rate2 : " + score
160 + " : " + EngineID
161 + " : " + AIEngine.GetName(EngineID)
162 + " : " + AIEngine.GetCapacity(EngineID)
163 + " * (" + AIEngine.GetReliability(EngineID) + " /100)"
164 + " * (" + AIEngine.GetMaxSpeed(EngineID) + " / 27 * 365)"
165 + " * " + capacity_factor
166 + " * 1000"
167 + " / (" + annualize_purchase.tointeger()
168 + " + " + AIEngine.GetRunningCost(EngineID)
169 +")",
170 7
171 );
172 return score;
173}

175function _MinchinWeb_Engine_::Rate3(EngineID, Cargo, MonthlyProduction, TravelDistance, PayDistance = 0, MaxSpend = 0) {
176 // Designed to Run as a validator
177 // Given the EngineID, it will score them; higher is better
178 // Note: Cargo doesn't fully work yet. Capacity is measured in the default
179 // cargo.
180
181 // payout = Capacity * Payout(Distance, time)
182 // payout{$} = Capacity{ton} * Payout(Distance, time){$/ton}
183 //
184 // time = LoadingTime + TravelTime * 2
185 // time = [Capacity / MonthlyProduction] + [ Distance / speed ] * 2
186 // time = [Capacity / MonthlyProduction] * + [ Distance / [speed] ] * 2
187 // time{days} = [Capacity{ton} / MonthlyProduction{ton/month}] * 30{day/month}
188 // + [ Distance{tiles} * 100{km/h}/3.6{tiles} / [speed{km/h}] ] * 2
189 //
190 // cost = (Purchase Price / Life) + (Running Costs)
191 // cost = (PurchasePrice{$} / Life{year}) + (RunningCosts{$/year})
192 // cost{$/day} = (PurchasePrice{$} / Life{year}) + (RunningCosts{$/year}) * {year}/365{day}
193 //
194 // score = payout / time - cost)
195
196 local _magnifier = 1000;
197 // only buildable engines
198 if (AIEngine.IsBuildable(EngineID)) {
199 // pass
200 } else {
201 return -1 * _MinchinWeb_C_.Infinity() * _magnifier;
202 }
203
204 // only return vehicles under our hard cost limit
205 if ((MaxSpend > 0) && (AIEngine.GetPrice(EngineID) > MaxSpend)) {
206 return -1 * _MinchinWeb_C_.Infinity() * _magnifier;
207 }
208
209 // only those that can carry our cargo
210 if (AIEngine.CanRefitCargo(EngineID, Cargo)) {
211 // pass
212 } else {
213 return -1 * _MinchinWeb_C_.Infinity() * _magnifier;
214 }
215
216 if (PayDistance == 0) {
217 PayDistance = TravelDistance;
218 }
219
220 local time = AIEngine.GetCapacity(EngineID).tofloat() / MonthlyProduction.tofloat() * 30;
221 time += (TravelDistance.tofloat() * 100.0 / 3.6 / AIEngine.GetMaxSpeed(EngineID).tofloat()) * 2;
222
223 // AIEngine.GetMaxAge() returns a value in (calendar) days
224 local _max_age = AIEngine.GetMaxAge(EngineID).tofloat();
225 local _daily_purchase = AIEngine.GetPrice(EngineID).tofloat() / _max_age;
226 local cost = _daily_purchase;
227 cost += AIEngine.GetRunningCost(EngineID).tofloat() / 365.0;
228
229 local payout = AICargo.GetCargoIncome(Cargo, PayDistance, time.tointeger());
230 payout *= AIEngine.GetCapacity(EngineID);
231 payout = payout.tofloat();
232
233 local score = payout / time - cost;
234 // scale up scores so integers can be used
235 score *= _magnifier;
236
237 score = score.tointeger();
238 _MinchinWeb_Log_.Note(
239 "Engine.Rate3 : " + score
240 + " : " + EngineID
241 + " : " + AIEngine.GetName(EngineID)
242 + " : " + AIEngine.GetCapacity(EngineID) + " tons"
243 + " : " + payout
244 + " / " + time
245 + " - " + cost
246 + " * " + _magnifier,
247 7
248 );
249 return score;
250}
