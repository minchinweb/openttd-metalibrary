272function _MinchinWeb_DLS_::GridPoints(End1, End2) {
273 // ## GridPoints
274 // Returns an array of TileIndexes that are 'grid points' or where roads
275 // will have intersections.
276 // *End1* and *End2* are expected to be TileIndex'es
277 // End1 and End2 will not be included in the return array
278
279 local x1 = AIMap.GetTileX(End1);
280 local y1 = AIMap.GetTileY(End1);
281 local x2 = AIMap.GetTileX(End2);
282 local y2 = AIMap.GetTileY(End2);
283
284 if (x1 > x2) {
285 local _temp_x = x1;
286 x1 = x2;
287 x2 = _temp_x;
288 }
289 if (y1 > y2) {
290 local _temp_y = y1;
291 y1 = y2;
292 y2 = _temp_y;
293 }
294
295 _MinchinWeb_Log_.Note("Generating grid points from " + x1 + ", " + y1 + " to " + x2 + ", " + y2, 6);
296
297 // move to first grid x
298 local _working_x = x1;
299 while ((_working_x - AIMap.GetTileX(this._datum)) % this._grid_x != 0) {
300 _working_x++;
301 }
302 // move to first grid y
303 local _working_y = y1;
304 while ((_working_y - AIMap.GetTileY(this._datum)) % this._grid_y != 0) {
305 _working_y++;
306 }
307
308 // Cycle through all the grid points and add them to our array
309 // use *do..while* to ensure we get at least one set of grid point per
310 // direction
311 local MyArray = [];
312 local _start_y = _working_y;
313 do {
314 do {
315 local Holding = AIMap.GetTileIndex(_working_x, _working_y);
316 if ((Holding != End1) && (Holding != End2)) {
317 MyArray.push(Holding);
318 _MinchinWeb_Log_.Note("Add grid point at " + _working_x + ", " + _working_y, 7)
319 }
320 _working_y = _working_y + this._grid_y;
321 } while (_working_y < y2);
322 _working_x = _working_x + this._grid_x;
323 _working_y = _start_y;
324 } while (_working_x < x2);
325
326 _MinchinWeb_Log_.Note("Generated " + MyArray.len() + " grid points.", 6);
327 return MyArray;
328}

335function _MinchinWeb_DLS_::FindPath(cycles = 10000) {
336 // runs the pathfinder
337 // add all grid points between the StartTile and EndTile as intermediate end points
338 // if the pathfinder ends on an intermediate point, make that the new start point and run the pathfinder again
339 local AllTiles = []; // we use this to return an array of tiles
340 local LastTile = this._start_tile;
341 cycles = 10000;
342 local StartArray = [];
343 local EndArray = [];
344 AllTiles.push(LastTile); // this adds our starting square to the array
345
346 local WhileLoopCounter = 0;
347
348 _MinchinWeb_Log_.Note("+ while loop before: LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 7);
349 do {
350 _MinchinWeb_Log_.Note("++ while loop start: LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 7);
351 StartArray = [LastTile];
352 EndArray = _MinchinWeb_DLS_.GridPoints(LastTile, this._end_tile);
353 EndArray.push(this._end_tile);
354 _MinchinWeb_Log_.Note("StartArray: " + _MinchinWeb_Array_.ToStringTiles1D(StartArray), 7);
355 _MinchinWeb_Log_.Note("EndArray: " + _MinchinWeb_Array_.ToStringTiles1D(EndArray), 7);
356
357 this._pathfinder = null; // dump old pathfinder
358 this._pathfinder = _MinchinWeb_RoadPathfinder_();
359 this._pathfinder.PresetQuickAndDirty();
360 this._pathfinder.InitializePath(EndArray, StartArray);
361 local Ret = this._pathfinder.FindPath(cycles);
362 this._running = (Ret == false) ? true : false;
363
364
365 _MinchinWeb_Log_.Note("AllTiles (before) : " + _MinchinWeb_Array_.ToStringTiles1D(AllTiles, true), 7);
366 local TilesToAdd = this._pathfinder.PathToTiles();
367 if ((typeof (TilesToAdd) == "array") && (TilesToAdd.len() > 0)) {
368 TilesToAdd.remove(0); // remove the first tile so we don't get duplicates
369 _MinchinWeb_Log_.Note("Path to Tiles: " + _MinchinWeb_Array_.ToStringTiles1D(TilesToAdd, true), 7);
370 AllTiles.extend(TilesToAdd);
371 }
372 _MinchinWeb_Log_.Note("AllTiles (1D): " + _MinchinWeb_Array_.ToStringTiles1D(AllTiles, true), 6);
373 LastTile = AllTiles.top();
374 _MinchinWeb_Log_.Note("+++ while loop: " + (LastTile != this._end_tile) + " && " + (this._running == true) + " = " + ((LastTile != this._end_tile) && (this._running == true)) + " ; LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 7);
375 _MinchinWeb_Log_.Note("LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 5);
376
377 WhileLoopCounter++;
378 // } while ((LastTile != this._end_tile) && (this._running == true));
379 } while (LastTile != this._end_tile)
380
381 if (LastTile == this._end_tile) {
382 this._path = AllTiles;
383 return true;
384 } else {
385 // I don't think we should ever get here...
386 // I guess it might get here if the pathfinder fails
387 return false;
388 }

399function _MinchinWeb_DLS_::BuildPath() {
400 if (this._running) {
401 AILog.Warning("You can't build a path while there's a running pathfinder.");
402 return false;
403 }
404 if (this._path == null) {
405 AILog.Warning("You have tried to build a 'null' path.");
406 return false;
407 }
408
409 local TestMode = AIExecMode(); // We're really doing this!
410
411 AIRoad.SetCurrentRoadType(this._road_type);
412 for (local i=0; i < this._path.len() - 2; i++) {
413 if (AIMap.DistanceManhattan(this._path[i], this._path[i+1]) == 1) {
414 // MD == 1 == road joining the two tiles
415 if (!AIRoad.BuildRoad(this._path[i], this._path[i+1])) {
416 // If we get here, then the road building has failed
417 // Possible that the road already exists
418 // TODO: fail the road builder if the road cannot be built and does not already exist
419 // return null;
420 }
421 } else {
422 // Implies that we're building either a tunnel or a bridge
423 if (!AIBridge.IsBridgeTile(this._path[i]) && !AITunnel.IsTunnelTile(this._path[i])) {
424 if (AIRoad.IsRoadTile(this._path[i])) {
425 // Original example demolishes tile if it's already a road
426 // tile to get around expanded roadbits.
427 // I don't like this approach as it could destroy Railway
428 // tracks/tram tracks/station
429 // TO-DO:
430 // - figure out a way to do this while keeping the other
431 // things I've built on the tile
432 // (can I just remove the road?)
433 AITile.DemolishTile(this._path[i]);
434 }
435 if (AITunnel.GetOtherTunnelEnd(this._path[i]) == this._path[i+1]) {
436 // The assumption here is that the land hasn't changed
437 // from when the pathfinder was run and when we try to
438 // build the path. If the tunnel building fails, we
439 // get the 'can't build tunnel' message, but if the
440 // land has changed such that the tunnel end is at a
441 // different spot than is was when the pathfinder ran,
442 // we skip tunnel building and try and build a bridge
443 // instead, which will fail because the slopes are wrong...
444 if (!AITunnel.BuildTunnel(AIVehicle.VT_ROAD, this._path[i])) {
445 // At this point, an error has occurred while building the tunnel.
446 // Fail the pathfiner
447 // return null;
448 AILog.Warning("MinchinWeb.DLS.BuildPath can't build a tunnel from " + AIMap.GetTileX(this._path[i]) + "," + AIMap.GetTileY(this._path[i]) + " to " + AIMap.GetTileX(this._path[i+1]) + "," + AIMap.GetTileY(this._path[i+1]) + "!!" );
449 }
450 } else {
451 // if not a tunnel, we assume we're building a bridge
452 local BridgeList = AIBridgeList_Length(AIMap.DistanceManhattan(this._path[i], this._path[i+1] + 1));
453 BridgeList.Valuate(AIBridge.GetMaxSpeed);
454 BridgeList.Sort(AIList.SORT_BY_VALUE, false);
455 if (!AIBridge.BuildBridge(AIVehicle.VT_ROAD, BridgeList.Begin(), this._path[i], this._path[i+1])) {
456 // At this point, an error has occurred while building the bridge.
457 // Fail the pathfiner
458 // return null;
459 AILog.Warning("MinchinWeb.DLS.BuildPath can't build a bridge from " + AIMap.GetTileX(this._path[i]) + "," + AIMap.GetTileY(this._path[i]) + " to " + AIMap.GetTileX(this._path[i+1]) + "," + AIMap.GetTileY(this._path[i+1]) + "!! (or the tunnel end moved...)" );
460 }
461 }
462 }
463 }
464 }
465
466 // End build sequence
467 return true;
