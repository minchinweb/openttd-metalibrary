584function _MinchinWeb_Array_::ToStringTiles1D(InArrayOfTiles, ArrayLength = false, replaceNull = false) {
585 if (InArrayOfTiles == null) {
586 return null;
587 } else {
588 local Length = InArrayOfTiles.len();
589 local Temp = "";
590 foreach (Tile in InArrayOfTiles) {
591 if ((replaceNull == true) && (Tile == null)) {
592 Temp = Temp + " ---";
593 } else if (Tile <= 256) {
594 // the first tile on the board is 0x0101, or 257 in decimal
595 Temp = Temp + " " + Tile;
596 } else {
597 Temp = Temp + " " + AIMap.GetTileX(Tile) + "," + AIMap.GetTileY(Tile);
598 }
599 }
600 if (ArrayLength == true) {
601 Temp = "The array is " + Length + " long. " + Temp;
602 }
603 return Temp;
