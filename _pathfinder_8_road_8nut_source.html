345function _MinchinWeb_RoadPathfinder_::_GetBridgeNumSlopes(end_a, end_b) {
346 local slopes = 0;
347 local direction = (end_b - end_a) / AIMap.DistanceManhattan(end_a, end_b);
348 local slope = AITile.GetSlope(end_a);
349 if (!((slope == AITile.SLOPE_NE && direction == 1) || (slope == AITile.SLOPE_SE && direction == -AIMap.GetMapSizeX()) ||
350 (slope == AITile.SLOPE_SW && direction == -1) || (slope == AITile.SLOPE_NW && direction == AIMap.GetMapSizeX()) ||
351 slope == AITile.SLOPE_N || slope == AITile.SLOPE_E || slope == AITile.SLOPE_S || slope == AITile.SLOPE_W)) {
352 slopes++;
353 }
354
355 local slope = AITile.GetSlope(end_b);
356 direction = -direction;
357 if (!((slope == AITile.SLOPE_NE && direction == 1) || (slope == AITile.SLOPE_SE && direction == -AIMap.GetMapSizeX()) ||
358 (slope == AITile.SLOPE_SW && direction == -1) || (slope == AITile.SLOPE_NW && direction == AIMap.GetMapSizeX()) ||
359 slope == AITile.SLOPE_N || slope == AITile.SLOPE_E || slope == AITile.SLOPE_S || slope == AITile.SLOPE_W)) {
360 slopes++;
361 }
362 return slopes;
363}

365function _MinchinWeb_RoadPathfinder_::_Cost(self, path, new_tile, new_direction) {
366 /* path == null means this is the first node of a path, so the cost is 0. */
367 if (path == null) return 0;
368
369 local prev_tile = path.GetTile();
370
371 /* If the new tile is (already) a bridge / tunnel tile, check whether we
372 * came from the other end of the bridge / tunnel or if we just entered the
373 * bridge / tunnel. */
374 if (AIBridge.IsBridgeTile(new_tile)) {
375 if (AIBridge.GetOtherBridgeEnd(new_tile) != prev_tile) {
376 return path.GetCost() + self._cost_tile;
377 } else {
378 return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * self._cost_tile + self._GetBridgeNumSlopes(new_tile, prev_tile) * self._cost_slope;
379 }
380 }
381 if (AITunnel.IsTunnelTile(new_tile)) {
382 if (AITunnel.GetOtherTunnelEnd(new_tile) != prev_tile) return path.GetCost() + self._cost_tile;
383 return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * self._cost_tile;
384 }
385
386 /* If the two tiles are more then 1 tile apart, the pathfinder wants a
387 * bridge or tunnel to be build. It isn't an existing bridge / tunnel, as
388 * that case is already handled. */
389 if (AIMap.DistanceManhattan(new_tile, prev_tile) > 1) {
390 /* Check if we should build a bridge or a tunnel. */
391 if (AITunnel.GetOtherTunnelEnd(new_tile) == prev_tile) {
392 return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * (self._cost_tile + self._cost_tunnel_per_tile);
393 } else {
394 return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * (self._cost_tile + self._cost_bridge_per_tile) + self._GetBridgeNumSlopes(new_tile, prev_tile) * self._cost_slope;
395 }
396 }
397
398 /* Check for a turn. We do this by substracting the TileID of the current node from
399 * the TileID of the previous node and comparing that to the difference between the
400 * previous node and the node before that. */
401 local cost = self._cost_tile;
402 if (path.GetParent() != null && (prev_tile - path.GetParent().GetTile()) != (new_tile - prev_tile) &&
403 AIMap.DistanceManhattan(path.GetParent().GetTile(), prev_tile) == 1) {
404 cost += self._cost_turn;
405 }
406
407 /* Check if the new tile is a coast tile. */
408 if (AITile.IsCoastTile(new_tile)) {
409 cost += self._cost_coast;
410 }
411
412 /* Check if the last tile was sloped. */
413 if (path.GetParent() != null && !AIBridge.IsBridgeTile(prev_tile) && !AITunnel.IsTunnelTile(prev_tile) &&
414 self._IsSlopedRoad(path.GetParent().GetTile(), prev_tile, new_tile)) {
415 cost += self._cost_slope;
416 }
417
418 /* Add a cost to "outcost" all paths that aren't using already existing
419 * roads, if that's what we're after */
420 if (!AIRoad.AreRoadTilesConnected(prev_tile, new_tile)) {
421 cost += self._cost_no_existing_road;
422 }
423
424 /* Add a penalty for road/rail level crossings. */
425 if(AITile.HasTransportType(new_tile, AITile.TRANSPORT_RAIL)) {
426 cost += self._cost_level_crossing;
427 }
428
429 /* Add a penalty for existing drive thru road stations */
430 if(AIRoad.IsDriveThroughRoadStationTile(new_tile)) {
431 cost += self._cost_drivethru_station;
432 }
433
434 return path.GetCost() + cost;
435}

437function _MinchinWeb_RoadPathfinder_::_Estimate(self, cur_tile, cur_direction, goal_tiles) {
438 local min_cost = self._max_cost;
439 /* As estimate we multiply the lowest possible cost for a single tile with
440 * with the minimum number of tiles we need to traverse. */
441 foreach (tile in goal_tiles) {
442 min_cost = min(AIMap.DistanceManhattan(cur_tile, tile) * self._cost_tile * self._distance_penalty, min_cost);
443 }
444 return min_cost;
445}

447function _MinchinWeb_RoadPathfinder_::_Neighbours(self, path, cur_node) {
448 /* self._max_cost is the maximum path cost, if we go over it, the path isn't valid. */
449 if (path.GetCost() >= self._max_cost) return [];
450 local tiles = [];
451
452 /* Check if the current tile is part of a bridge or tunnel. */
453 if ((AIBridge.IsBridgeTile(cur_node) || AITunnel.IsTunnelTile(cur_node)) &&
454 AITile.HasTransportType(cur_node, AITile.TRANSPORT_ROAD)) {
455 local other_end = AIBridge.IsBridgeTile(cur_node) ? AIBridge.GetOtherBridgeEnd(cur_node) : AITunnel.GetOtherTunnelEnd(cur_node);
456 local next_tile = cur_node + (cur_node - other_end) / AIMap.DistanceManhattan(cur_node, other_end);
457 if (AIRoad.AreRoadTilesConnected(cur_node, next_tile) || AITile.IsBuildable(next_tile) || AIRoad.IsRoadTile(next_tile)) {
458 tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
459 }
460 /* The other end of the bridge / tunnel is a neighbour. */
461 tiles.push([other_end, self._GetDirection(next_tile, cur_node, true) << 4]);
462 } else if (path.GetParent() != null && AIMap.DistanceManhattan(cur_node, path.GetParent().GetTile()) > 1) {
463 local other_end = path.GetParent().GetTile();
464 local next_tile = cur_node + (cur_node - other_end) / AIMap.DistanceManhattan(cur_node, other_end);
465 if (AIRoad.AreRoadTilesConnected(cur_node, next_tile) || AIRoad.BuildRoad(cur_node, next_tile)) {
466 tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
467 }
468 } else {
469 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
470 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
471 /* Check all tiles adjacent to the current tile. */
472 foreach (offset in offsets) {
473 local next_tile = cur_node + offset;
474 /* We add them to the to the neighbours-list if one of the following applies:
475 * 1) There already is a connections between the current tile and the next tile.
476 * 2) We can build a road to the next tile.
477 * 3) The next tile is the entrance of a tunnel / bridge in the correct direction. */
478 if (AIRoad.AreRoadTilesConnected(cur_node, next_tile)) {
479 tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
480 } else if ((self._cost_only_existing_roads != true) && (AITile.IsBuildable(next_tile) || AIRoad.IsRoadTile(next_tile)) &&
481 (path.GetParent() == null || AIRoad.CanBuildConnectedRoadPartsHere(cur_node, path.GetParent().GetTile(), next_tile)) &&
482 AIRoad.BuildRoad(cur_node, next_tile)) {
483 // WM - add '&& (only_existing_roads != true)' so that non-connected roads are ignored
484 tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
485 } else if ((self._cost_only_existing_roads != true) && self._CheckTunnelBridge(cur_node, next_tile)) {
486 tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
487 }
488
489 // Test for water (i.e. rivers or canals or rails to bridge over them
490 local iTile = cur_node + offset;
491 local BridgeLength = 2;
492 while (AITile.HasTransportType(iTile, AITile.TRANSPORT_RAIL) || AITile.IsWaterTile(iTile)) {
493 iTile += offset;
494 BridgeLength++;
495 }
496
497 // test to see if we could actually build said bridge
498 // TO-DO: Check to see if this test is done elsewhere...
499
500 if (BridgeLength > 2) {
501 // TO-DO: test for map wraparound... _SuperLib_Tile::IsStraight(tile1, tile2)
502 local BridgeList = AIBridgeList_Length(BridgeLength);
503 if ((BridgeList.Count()) > 0 && (AIBridge.BuildBridge(AIVehicle.VT_ROAD, BridgeList.Begin(), cur_node, iTile))) {
504 local PathCheck = path;
505 local PathParent = path.GetParent();
506 // _MinchinWeb_Log_.Note("Adding Bridge-over tile: " + _MinchinWeb_Array_.ToStringTiles1D([cur_node]) + _MinchinWeb_Array_.ToStringTiles1D([iTile]) + " . " + (self._GetDirection(iTile, cur_node, true) << 4), 7);
507 tiles.push([iTile, self._GetDirection(iTile, cur_node, true) << 4]);
508 }
509 }
510
511 }
512 if (path.GetParent() != null) {
513 local bridges = self._GetTunnelsBridges(path.GetParent().GetTile(), cur_node, self._GetDirection(path.GetParent().GetTile(), cur_node, true) << 4);
514 foreach (tile in bridges) {
515 tiles.push(tile);
516 }
517 }
518 }
519 return tiles;
520}

526function _MinchinWeb_RoadPathfinder_::_GetDirection(from, to, is_bridge) {
527 if (!is_bridge && AITile.GetSlope(to) == AITile.SLOPE_FLAT) return 0xFF;
528 if (from - to == 1) return 1;
529 if (from - to == -1) return 2;
530 if (from - to == AIMap.GetMapSizeX()) return 4;
531 if (from - to == -AIMap.GetMapSizeX()) return 8;
532
533 // for bridges that don't have a parent tile
534 local direction = from - to;
535 if (direction > 0) {
536 // so direction is positive
537 if (direction < (AIMap.GetMapSizeX() / 2 - 1)) return 1;
538 else return 4;
539 } else {
540 if ((direction * -1) < (AIMap.GetMapSizeX() / 2 - 1)) return 2;
541 else return 8;
542 }
543}

551function _MinchinWeb_RoadPathfinder_::_GetTunnelsBridges(last_node, cur_node, bridge_dir) {
552 // By rights, adding bridge over railroads and water should be added here
553 local slope = AITile.GetSlope(cur_node);
554 if (slope == AITile.SLOPE_FLAT) return [];
555 local tiles = [];
556
557 for (local i = 2; i < this._max_bridge_length; i++) {
558 local bridge_list = AIBridgeList_Length(i + 1);
559 local target = cur_node + i * (cur_node - last_node);
560 if (!bridge_list.IsEmpty() && AIBridge.BuildBridge(AIVehicle.VT_ROAD, bridge_list.Begin(), cur_node, target)) {
561 tiles.push([target, bridge_dir]);
562 }
563 }
564
565 if (slope != AITile.SLOPE_SW && slope != AITile.SLOPE_NW && slope != AITile.SLOPE_SE && slope != AITile.SLOPE_NE) return tiles;
566 local other_tunnel_end = AITunnel.GetOtherTunnelEnd(cur_node);
567 if (!AIMap.IsValidTile(other_tunnel_end)) return tiles;
568
569 local tunnel_length = AIMap.DistanceManhattan(cur_node, other_tunnel_end);
570 local prev_tile = cur_node + (cur_node - other_tunnel_end) / tunnel_length;
571 if (AITunnel.GetOtherTunnelEnd(other_tunnel_end) == cur_node && tunnel_length >= 2 &&
572 prev_tile == last_node && tunnel_length < _max_tunnel_length && AITunnel.BuildTunnel(AIVehicle.VT_ROAD, cur_node)) {
573 tiles.push([other_tunnel_end, bridge_dir]);
574 }
575 return tiles;
576}

578function _MinchinWeb_RoadPathfinder_::_IsSlopedRoad(start, middle, end) {
579 local NW = 0; //Set to true if we want to build a road to / from the north-west
580 local NE = 0; //Set to true if we want to build a road to / from the north-east
581 local SW = 0; //Set to true if we want to build a road to / from the south-west
582 local SE = 0; //Set to true if we want to build a road to / from the south-east
583
584 if (middle - AIMap.GetMapSizeX() == start || middle - AIMap.GetMapSizeX() == end) NW = 1;
585 if (middle - 1 == start || middle - 1 == end) NE = 1;
586 if (middle + AIMap.GetMapSizeX() == start || middle + AIMap.GetMapSizeX() == end) SE = 1;
587 if (middle + 1 == start || middle + 1 == end) SW = 1;
588
589 /* If there is a turn in the current tile, it can't be sloped. */
590 if ((NW || SE) && (NE || SW)) return false;
591
592 local slope = AITile.GetSlope(middle);
593 /* A road on a steep slope is always sloped. */
594 if (AITile.IsSteepSlope(slope)) return true;
595
596 /* If only one corner is raised, the road is sloped. */
597 if (slope == AITile.SLOPE_N || slope == AITile.SLOPE_W) return true;
598 if (slope == AITile.SLOPE_S || slope == AITile.SLOPE_E) return true;
599
600 if (NW && (slope == AITile.SLOPE_NW || slope == AITile.SLOPE_SE)) return true;
601 if (NE && (slope == AITile.SLOPE_NE || slope == AITile.SLOPE_SW)) return true;
602
603 return false;
604}

606function _MinchinWeb_RoadPathfinder_::_CheckTunnelBridge(current_tile, new_tile) {
607 if (!AIBridge.IsBridgeTile(new_tile) && !AITunnel.IsTunnelTile(new_tile)) return false;
608 local dir = new_tile - current_tile;
609 local other_end = AIBridge.IsBridgeTile(new_tile) ? AIBridge.GetOtherBridgeEnd(new_tile) : AITunnel.GetOtherTunnelEnd(new_tile);
610 local dir2 = other_end - new_tile;
611 if ((dir < 0 && dir2 > 0) || (dir > 0 && dir2 < 0)) return false;
612 dir = abs(dir);
613 dir2 = abs(dir2);
614 if ((dir >= AIMap.GetMapSizeX() && dir2 < AIMap.GetMapSizeX()) ||
615 (dir < AIMap.GetMapSizeX() && dir2 >= AIMap.GetMapSizeX())) return false;
616
617 return true;
618}

753function _MinchinWeb_RoadPathfinder_::GetBuildCost() {
754 // Turns to 'test mode,' builds the route provided, and returns the cost
755 // (all money for AI's is in British Pounds)
756 // Note that due to inflation, this value can get stale
757 // Returns false if the test build fails somewhere
758
759 if (this._running) {
760 AILog.Warning("You can't find the build costs while there's a running pathfinder.");
761 return false;
762 }
763 if (this._my_path == null) {
764 AILog.Warning("You have tried to get the build costs of a 'null' path.");
765 return false;
766 }
767
768 local BeanCounter = AIAccounting();
769 local TestMode = AITestMode();
770 local Path = this._my_path;
771
772 AIRoad.SetCurrentRoadType(this._road_type);
773 while (Path != null) {
774 local SubPath = Path.GetParent();
775 if (SubPath != null) {
776 local Node = Path.GetTile();
777 if (AIMap.DistanceManhattan(Path.GetTile(), SubPath.GetTile()) == 1) {
778 // MD == 1 == road joining the two tiles
779 if (!AIRoad.BuildRoad(Path.GetTile(), SubPath.GetTile())) {
780 // If we get here, then the road building has failed
781 // Possible that the road already exists
782 // TO-DO
783 // - fail the road builder if the road cannot be built and
784 // does not already exist
785 // return null;
786 }
787 } else {
788 // Implies that we're building either a tunnel or a bridge
789 if (!AIBridge.IsBridgeTile(Path.GetTile()) && !AITunnel.IsTunnelTile(Path.GetTile())) {
790 if (AIRoad.IsRoadTile(Path.GetTile())) {
791 // Original example demolishes tile if it's already a road
792 // tile to get around expanded roadbits.
793 // I don't like this approach as it could destroy
794 // Railway tracks/tram tracks/station
795 // TO-DO
796 // - figure out a way to do this while keeping the
797 // other things I've built on the tile
798 // (can I just remove the road?)
799 AITile.DemolishTile(Path.GetTile());
800 }
801 if (AITunnel.GetOtherTunnelEnd(Path.GetTile()) == SubPath.GetTile()) {
802 if (!AITunnel.BuildTunnel(AIVehicle.VT_ROAD, Path.GetTile())) {
803 // At this point, an error has occurred while
804 // building the tunnel.
805 // Fail the pathfinder
806 // return null;
807 AILog.Warning("MinchinWeb.RoadPathfinder.GetBuildCost can't build a tunnel from " + AIMap.GetTileX(Path.GetTile()) + "," + AIMap.GetTileY(Path.GetTile()) + " to " + AIMap.GetTileX(SubPath.GetTile()) + "," + AIMap.GetTileY(SubPath.GetTile()) + "!!" );
808 }
809 } else {
810 // if not a tunnel, we assume we're building a bridge
811 local BridgeList = AIBridgeList_Length(AIMap.DistanceManhattan(Path.GetTile(), SubPath.GetTile() + 1));
812 BridgeList.Valuate(AIBridge.GetMaxSpeed);
813 BridgeList.Sort(AIList.SORT_BY_VALUE, false);
814 if (!AIBridge.BuildBridge(AIVehicle.VT_ROAD, BridgeList.Begin(), Path.GetTile(), SubPath.GetTile())) {
815 // At this point, an error has occurred while
816 // building the bridge.
817 // Fail the pathfinder
818 // return null;
819 AILog.Warning("MinchinWeb.RoadPathfinder.GetBuildCost can't build a bridge from " + AIMap.GetTileX(Path.GetTile()) + "," + AIMap.GetTileY(Path.GetTile()) + " to " + AIMap.GetTileX(SubPath.GetTile()) + "," + AIMap.GetTileY(SubPath.GetTile()) + "!!" );
820 }
821 }
822 }
823 }
824 }
825 Path = SubPath;
826 }
827
828 // End build sequence
829 return BeanCounter.GetCosts();
830}
