238function _MinchinWeb_Extras_::SignLocation(text) {

249function _MinchinWeb_Extras_::MidPoint(TileA, TileB) {
250 local X = (AIMap.GetTileX(TileA) + AIMap.GetTileX(TileB)) / 2 + 0.5;
251 local Y = (AIMap.GetTileY(TileA) + AIMap.GetTileY(TileB)) / 2 + 0.5;
252 // the 0.5 is to make rounding work
253 X = X.tointeger();
254 Y = Y.tointeger();
255 return AIMap.GetTileIndex(X, Y);
256}

267function _MinchinWeb_Extras_::Slope(TileA, TileB) {
268 local dx = AIMap.GetTileX(TileB) - AIMap.GetTileX(TileA);
269 local dy = AIMap.GetTileY(TileB) - AIMap.GetTileY(TileA);
270
271 // Zero check
272 if (dx == 0) {
273 return _MinchinWeb_C_.Infinity() * _MinchinWeb_Extras_.Sign(dy);
274 } else if (dy == 0) {
275 return (1.0 / _MinchinWeb_C_.Infinity()) * _MinchinWeb_Extras_.Sign(dx);
276 } else {
277 dx = dx.tofloat();
278 dy = dy.tofloat();
279
280 return (dy / dx);
281 }
282}

291function _MinchinWeb_Extras_::WithinFloat(Bound1, Bound2, Value) {
292 local UpperBound = _MinchinWeb_Extras_.MaxFloat(Bound1, Bound2) + _MinchinWeb_C_.FloatOffset();
293 local LowerBound = _MinchinWeb_Extras_.MinFloat(Bound1, Bound2) - _MinchinWeb_C_.FloatOffset();
294 local Value = Value.tofloat();
295
296// _MinchinWeb_Log_.Note(" Extras.WithinFloat: Val=" + Value + " B1=" + Bound1 + " B2=" + Bound2 + " : UB=" + UpperBound + " LB=" + LowerBound + " is " + (Value <= UpperBound) + " " + (Value >= LowerBound) + " : " + ((Value <= UpperBound) && (Value >= LowerBound)) + " : above " + (Value - UpperBound) + " below " + (LowerBound - Value) + " : " + _MinchinWeb_C_.FloatOffset() , 7);
297
298 return ((Value <= UpperBound) && (Value >= LowerBound));

377function _MinchinWeb_Extras_::NextCardinalTile(StartTile, TowardsTile) {
378 local Tiles = AITileList();
379 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
380 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
381
382 foreach (offset in offsets) {
383 Tiles.AddItem(StartTile + offset, AIMap.DistanceSquare(StartTile + offset, TowardsTile));
384 }
385
386 Tiles.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
387
388 return Tiles.Begin();
389}

397function _MinchinWeb_Extras_::MinDistance(TileID, TargetArray) {
398 local MinDist = _MinchinWeb_C_.Infinity();
399 foreach (Target in TargetArray) {
400 // note that with AIList's, Target is the `Value`, not the `Item`
401 MinDist = min(MinDist, AITile.GetDistanceManhattanToTile(TileID, Target));
402 }
403 return MinDist;
404}
