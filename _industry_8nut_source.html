48function _MinchinWeb_Industry_::GetIndustryID(Tile) {
49 local StartX = AIMap.GetTileX(Tile);
50 local StartY = AIMap.GetTileY(Tile);
51 local EndX = AIMap.GetTileX(Tile) + _MinchinWeb_C_.IndustrySize();
52 local EndY = AIMap.GetTileY(Tile) + _MinchinWeb_C_.IndustrySize();
53
54 for (local i = StartX; i < EndX; i++) {
55 for (local j = StartY; j < EndY; j++) {
56 if (AIIndustry.GetIndustryID(AIMap.GetTileIndex(i,j)) != _MinchinWeb_C_.InvalidIndustry()) {
57 return AIIndustry.GetIndustryID(AIMap.GetTileIndex(i,j));
58 }
59 }
60 }
61
62 // if no valid industry is found...
63 return _MinchinWeb_C_.InvalidIndustry();
64}
