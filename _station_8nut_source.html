97function _MinchinWeb_Station_::IsCargoAccepted(StationID, CargoID) {
98 if (!AIStation.IsValidStation(StationID) || !AICargo.IsValidCargo(CargoID)) {
99 AILog.Warning("MinchinWeb.Station.IsCargoAccepted() was provided with invalid input. Was provided " + StationID + " and " + CargoID + ".");
100 return null;
101 } else {
102 local AllCargos = AICargoList_StationAccepting(StationID);
103 _MinchinWeb_Log_.Note("MinchinWeb.Station.IsCargoAccepted() was provided with " + StationID + " and " + CargoID + ". AllCargos: " + AllCargos.Count(), 6);
104 if (AllCargos.HasItem(CargoID)) {
105 return true;
106 } else {
107 return false;
108 }
109 }
110}

112function _MinchinWeb_Station_::IsNextToDock(TileID) {
113 local offsets = [0, AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
114 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
115
116 foreach (offset in offsets) {
117 if (AIMarine.IsDockTile(TileID + offset)) {
118 return true;
119 }
120 }
121
122 return false;

132function _MinchinWeb_Station_::BuildStreetcarStation(Tile, Loop = true) {
133 local TestMode = AITestMode();
134 AIRoad.SetCurrentRoadType(AIRoad.ROADTYPE_TRAM);
135 local FrontTile;
136 local BackTile;
137 local MyDirection;
138
139 // determine our road directions
140 if (AIRoad.BuildDriveThroughRoadStation(
141 Tile,
142 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_NE),
143 AIRoad.ROADVEHTYPE_BUS,
144 AIStation.STATION_NEW
145 ) && AIRoad.BuildRoad(
146 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_NE),
147 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_SW))
148 ) {
149 MyDirection = SuperLib.Direction.DIR_NE;
150 } else if (AIRoad.BuildDriveThroughRoadStation(
151 Tile,
152 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_SE),
153 AIRoad.ROADVEHTYPE_BUS,
154 AIStation.STATION_NEW
155 ) && AIRoad.BuildRoad(
156 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_SE),
157 SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.DIR_NW))
158 ) {
159 MyDirection = SuperLib.Direction.DIR_SE;
160 } else {
161 return false;
162 }
163
164 // get the tiles in front and behind our proposed station
165 FrontTile = SuperLib.Direction.GetAdjacentTileInDirection(Tile, MyDirection);
166 BackTile = SuperLib.Direction.GetAdjacentTileInDirection(Tile, SuperLib.Direction.OppositeDir(MyDirection));
167
168 local Pathfinder = _MinchinWeb_RoadPathfinder_();
169 if (Loop) {
170 Pathfinder.InitializePath([FrontTile], [BackTile], [Tile]);
171 Pathfinder.PresetStreetcar();
172 local pf_return = Pathfinder.FindPath(50000);
173 if ((pf_return == null) || (pf_return == false)) {
174 // the Pathfinder will return `false` if it is still looking, and
175 // `null` if it can't find a valid path
176 Log.Note("No loop path." + _MinchinWeb_Array_.ToStringTiles1D([Tile]), 7);
177 return false;
178 }
179 }
180
181 // Build for real!
182 local ExecMode = AIExecMode();
183 if (AIRoad.BuildRoad(FrontTile, BackTile)) {
184 // we keep doing stuff
185 AIRoad.BuildDriveThroughRoadStation(Tile, SuperLib.Direction.GetAdjacentTileInDirection(Tile, MyDirection), AIRoad.ROADVEHTYPE_BUS, AIStation.STATION_NEW);
186 if (Loop) {
187 SuperLib.Money.MakeSureToHaveAmount(Pathfinder.GetBuildCost());
188 Pathfinder.BuildPath();
189 }
190
191 return true;
192 } else {
193 // TODO: if road building fails on one direction, try the other
194 Log.Note("Streetcar Stations:" + _MinchinWeb_Array_.ToStringTiles1D([Tile]) + " Our little road building failed... exiting", 7);
195 return false;
196 }
197}

200function _MinchinWeb_Station_::GetName(Tile, StationType = AIStation.STATION_ANY) {
201 local station_list = AIStationList(StationType);
202 station_list.Valuate(AIStation.GetDistanceManhattanToTile, Tile);
203 station_list.KeepValue(0);
204 local station_name = AIBaseStation.GetName(station_list.Begin());
205
206 return station_name;
207}
