162function _MinchinWeb_SW_::Walk() {
163 if (this._Steps == 0) {
164 this._Steps++;
165 } else {
166
167 this._x += this._dx;
168 this._y += this._dy;
169 this._StageSteps ++;
170 this._Steps ++;
171
172 // Check if it's time to turn
173 if (this._StageSteps == this._StageMax) {
174 this._StageSteps = 0;
175 if (this._Stage % 2 == 0) {
176 this._StageMax++;
177 }
178 this._Stage ++;
179
180 // Turn Clockwise
181 switch (this._dx) {
182 case 0:
183 switch (this._dy) {
184 case -1:
185 this._dx = -1;
186 this._dy = 0;
187 break;
188 case 1:
189 this._dx = 1;
190 this._dy = 0;
191 break;
192 }
193 break;
194 case -1:
195 this._dx = 0;
196 this._dy = 1;
197 break;
198 case 1:
199 this._dx = 0;
200 this._dy = -1;
201 break;
202 }
203 }
204 }
205 _MinchinWeb_Log_.Note(
206 "SpiralWalker.Walk: " + this._dx + " " + this._dy + " : "
207 + this._Steps + " // " + this._Stage + " / " + this._StageSteps + " / "
208 + this._StageMax + " :: " + this._x + ", " + this._y,
209 8
210 );
211 this._current_tile = AIMap.GetTileIndex(this._x, this._y);
212// AISign.BuildSign(this._current_tile, "" + this._Steps);
213 return this._current_tile;
214}
