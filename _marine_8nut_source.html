229function _MinchinWeb_Marine_::DistanceShip(TileA, TileB) {
230// Assuming open ocean, ship in OpenTTD will travel 45° angle where possible,
231// and then finish up the trip by going along a cardinal direction
232 return ((AIMap.DistanceManhattan(TileA, TileB) - AIMap.DistanceMax(TileA, TileB)) * 0.4 + AIMap.DistanceMax(TileA, TileB)).tointeger();
233}

235function _MinchinWeb_Marine_::GetPossibleDockTiles(IndustryID) {
236// Given an industry (by IndustryID), searches for possible tiles to build a
237// dock and returns the list as an array of TileIndexs
238
239// Tiles given should be checked to ensure that the desired cargo is still
240// accepted
241
242// Assumes that the industry location returned is the NE corner of the
243// industry, and that industries fit within a 4x4 block
244 local Tiles = [];
245 if (AIIndustry.IsValidIndustry(IndustryID) == true) {
246 // Check if the industry already has a dock
247 if (AIIndustry.HasDock(IndustryID) == true) {
248 return [AIIndustry.GetDockLocation(IndustryID)];
249 } else {
250 local ex = AITestMode();
251 local Walker = _MinchinWeb_SW_(); // Spiral Walker
252 Walker.Start(AIIndustry.GetLocation(IndustryID));
253
254 while (Walker.GetStage() <= ((_MinchinWeb_C_.IndustrySize() + AIStation.GetCoverageRadius(AIStation.STATION_DOCK)) * 4)) {
255 if (AIMarine.BuildDock(Walker.Walk(), AIStation.STATION_NEW) == true) {
256 Tiles.push(Walker.GetTile());
257 }
258 }
259 }
260 _MinchinWeb_Log_.Note("MinchinWeb.Marine.GetPossibleDockTiles() " + _MinchinWeb_Array_.ToStringTiles1D(Tiles, true), 6);
261 return Tiles;
262 } else {
263 AILog.Warning("MinchinWeb.Marine.GetPossibleDockTiles() was supplied with an invalid IndustryID. Was supplied " + IndustryID + ".");
264 return Tiles;
265 }

268function _MinchinWeb_Marine_::GetDockFrontTiles(Tile) {
269// Given a tile, returns an array of possible 'front' tiles that a ship could
270// access the dock from
271// Can be either the land tile of a dock, or the water tile
272// Does not test if there is currently a dock at the tile
273
274// Tiles under Oil Rigs do NOT return AITile.IsWaterTile(Tile) == true
275
276// Might do funny things if the tile given is next to a river (i.e. a flat tile
277// next to a water tile)
278
279 local ReturnTiles = [];
280 local offset = AIMap.GetTileIndex(0, 0);;
281 local DockEnd = null;
282 local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
283 AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
284 local next_tile;
285
286 if (AIMap.IsValidTile(Tile)) {
287 if (AITile.IsWaterTile(Tile)) {
288 // water tile
289 DockEnd = Tile;
290 } else {
291 // land tile
292 switch (AITile.GetSlope(Tile)) {
293 // see http://vcs.openttd.org/svn/browser/trunk/docs/tileh.png
294 // for slopes
295 case 0:
296 // flat
297 offset = AIMap.GetTileIndex(0, 0);
298 break;
299 case 3:
300 offset = AIMap.GetTileIndex(-1, 0);
301 break;
302 case 6:
303 offset = AIMap.GetTileIndex(0, -1);
304 break;
305 case 9:
306 offset = AIMap.GetTileIndex(0, 1);
307 break;
308 case 12:
309 offset = AIMap.GetTileIndex(1, 0);
310 break;
311 }
312
313 DockEnd = Tile + offset;
314 }
315
316 if (DockEnd != null) {
317 /* Check all tiles adjacent to the current tile. */
318 foreach (offset in offsets) {
319 next_tile = DockEnd + offset;
320 if (AITile.IsWaterTile(next_tile)) {
321 ReturnTiles.push(next_tile);
322 }
323 }
324 }
325 } else {
326 AILog.Warning("MinchinWeb.Marine.GetDockFrontTiles() was supplied with an invalid TileIndex. Was supplied " + Tile + ".");
327 }
328
329 return ReturnTiles;
330}

332function _MinchinWeb_Marine_::BuildBuoy(Tile) {
333 // Attempts to build a buoy, but first checks the box within
334 // MinchinWeb.Constants.BuoyOffset() for an existing buoy, and makes sure
335 // there's nothing but water between the two. If no existing buoy is found,
336 // one is built.
337
338 // Returns the location of the existing or built buoy.
339
340 local Existing = AITileList();
341 local UseExistingAt = null;
342
343 local Walker = _MinchinWeb_SW_(); // Spiral Walker
344 Walker.Start(Tile);
345
346 while (Walker.GetStage() <= (_MinchinWeb_C_.BuoyOffset() * 4)) {
347 if (AIMarine.IsBuoyTile(Walker.Walk())) {
348 Existing.AddItem(Walker.GetTile(), AIMap.DistanceManhattan(Tile, Walker.GetTile()));
349 _MinchinWeb_Log_.Note("BuildBuoy() : Insert Existing at" + _MinchinWeb_Array_.ToStringTiles1D([Walker.GetTile()]), 7);
350 }
351 }
352
353 if (Existing.Count() > 0) {
354 Existing.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
355 local TestBuoy = Existing.Begin();
356 local KeepTrying = true;
357 local WBC = _MinchinWeb_WBC_();
358 while (KeepTrying == true) {
359 WBC.InitializePath([TestBuoy], [Tile]);
360 WBC.PresetSafety(TestBuoy, Tile);
361 local WBCResults = WBC.FindPath(-1);
362 if (WBCResults != null) {
363 UseExistingAt = TestBuoy;
364 KeepTrying = false;
365 } else {
366 if (Existing.IsEnd()) {
367 KeepTrying = false;
368 UseExistingAt = null;
369 } else {
370 TestBuoy = Existing.Next();
371 }
372 }
373 }
374 }
375
376 if (UseExistingAt == null) {
377 AIMarine.BuildBuoy(Tile);
378 return Tile;
379 } else {
380 return UseExistingAt;
381 }

384function _MinchinWeb_Marine_::BuildDepot(DockTile, Front, NotNextToDock=true) {
385 // Attempts to build a (water) depot, but first checks the box within
386 // MinchinWeb.Constants.WaterDepotOffset() for an existing depot, and makes
387 // sure there's nothing but water between the depot and dock. If no
388 // existing depot is found, one is built.
389
390 // Returns the location of the existing or built depot.
391 // This will fail if the DockTile given is a dock (or any tile that is not a water tile)
392
393 // 'NotNextToDock,' when set, will keep the dock from being built next to an
394 // existing dock
395
396 local StartX = AIMap.GetTileX(DockTile) - _MinchinWeb_C_.WaterDepotOffset();
397 local StartY = AIMap.GetTileY(DockTile) - _MinchinWeb_C_.WaterDepotOffset();
398 local EndX = AIMap.GetTileX(DockTile) + _MinchinWeb_C_.WaterDepotOffset();
399 local EndY = AIMap.GetTileY(DockTile) + _MinchinWeb_C_.WaterDepotOffset();
400
401 local Existing = AITileList();
402 local UseExistingAt = null;
403
404 for (local i = StartX; i < EndX; i++) {
405 for (local j = StartY; j < EndY; j++) {
406 if (AIMarine.IsWaterDepotTile(AIMap.GetTileIndex(i,j))) {
407 Existing.AddItem(AIMap.GetTileIndex(i,j), AIMap.DistanceManhattan(DockTile, AIMap.GetTileIndex(i,j)));
408 _MinchinWeb_Log_.Note("BuildDepot() : Insert Existing at" + _MinchinWeb_Array_.ToStringTiles1D([AIMap.GetTileIndex(i,j)]), 6);
409 }
410 }
411 }
412
413 if (Existing.Count() > 0) {
414 Existing.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
415 local TestDepot = Existing.Begin();
416 local KeepTrying = true;
417 local WBC = _MinchinWeb_WBC_();
418 while (KeepTrying == true) {
419 WBC.InitializePath([TestDepot], [DockTile]);
420 WBC.PresetSafety(TestDepot, DockTile);
421 local WBCResults = WBC.FindPath(-1);
422 if (WBCResults != null) {
423 UseExistingAt = TestDepot;
424 KeepTrying = false;
425 } else {
426 if (Existing.IsEnd()) {
427 KeepTrying = false;
428 UseExistingAt = null;
429 } else {
430 TestDepot = Existing.Next();
431 }
432 }
433 }
434 }
435
436 if (UseExistingAt == null) {
437 if(AIMarine.BuildWaterDepot(DockTile, Front)) {
438 // try and build right at the given spot
439 UseExistingAt = DockTile;
440 } else {
441 // if that doesn't work, build it close by
442 // Generate a list of water tiles, and pick one at random
443 Existing.Clear();
444 for (local i = StartX; i < EndX; i++) {
445 for (local j = StartY; j < EndY; j++) {
446 if (AITile.IsWaterTile(AIMap.GetTileIndex(i,j)) && (_MinchinWeb_Station_.IsNextToDock(AIMap.GetTileIndex(i,j)) == false) ) {
447 Existing.AddItem(AIMap.GetTileIndex(i,j), AIBase.Rand());
448 _MinchinWeb_Log_.Note("BuildDepot() : Insert WaterTile at" + _MinchinWeb_Array_.ToStringTiles1D([AIMap.GetTileIndex(i,j)]), 7);
449 }
450 }
451 }
452
453 Existing.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
454 local TestDepot = Existing.Begin();
455 local KeepTrying = true;
456 local WBC = _MinchinWeb_WBC_();
457 while (KeepTrying == true) {
458 WBC.InitializePath([TestDepot], [DockTile]);
459 WBC.PresetSafety(TestDepot, DockTile);
460 local WBCResults = WBC.FindPath(-1);
461 _MinchinWeb_Log_.Note("BuildDepot() : WBC on" + _MinchinWeb_Array_.ToStringTiles1D([TestDepot, DockTile]) + " returned " + WBCResults, 7);
462 if (WBCResults != null) {
463 local Front2 = _MinchinWeb_Extras_.NextCardinalTile(TestDepot, DockTile);
464 if (AIMarine.BuildWaterDepot(TestDepot, Front2))
465 {
466 UseExistingAt = TestDepot;
467 KeepTrying = false;
468 } else {
469 if (Existing.IsEnd()) {
470 KeepTrying = false;
471 UseExistingAt = null;
472 } else {
473 TestDepot = Existing.Next();
474 }
475 }
476 } else {
477 if (Existing.IsEnd()) {
478 KeepTrying = false;
479 UseExistingAt = null;
480 } else {
481 TestDepot = Existing.Next();
482 }
483 }
484 }
485
486 }
487 }
488
489 return UseExistingAt;

492function _MinchinWeb_Marine_::RateShips(EngineID, Life, Cargo) {
493 // Designed to Run as a validator
494 // Given the EngineID, it will score them; higher is better
495 // Score = [(Capacity in Cargo) * Reliability * Speed] /
496 // [ (Purchase Price over Life) + (Running Costs) * Life ]
497 //
498 // Life is assumed to be in years
499 // Note: Cargo doesn't work yet. Capacity is measured in the default cargo.
500
501 local Score = 0;
502 local Age = AIEngine.GetMaxAge(EngineID);
503 local BuyTimes = (Life / Age/366).tointeger() + 1;
504 // GetMaxAge is given in days
505 local Cost = (BuyTimes * AIEngine.GetPrice(EngineID)) + (Life * AIEngine.GetRunningCost(EngineID)) + 0.001;
506 local Return = (AIEngine.GetCapacity(EngineID) * AIEngine.GetReliability(EngineID) * AIEngine.GetMaxSpeed(EngineID)) + 0.001;
507 if (Return == 0) {
508 Score = 0;
509 } else {
510 Score = (Return * 1000 / Cost).tointeger();
511 }
512
513 _MinchinWeb_Log_.Note("Rate Ship : " + Score + " : " + AIEngine.GetName(EngineID) + " : " + AIEngine.GetCapacity(EngineID) + " * " + AIEngine.GetReliability(EngineID) + " * " + AIEngine.GetMaxSpeed(EngineID) + " / " + BuyTimes + " * " + AIEngine.GetPrice(EngineID) + " + " + Life + " * " + AIEngine.GetRunningCost(EngineID), 7);
514 return Score;
515}
